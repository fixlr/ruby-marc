    def each 
      @handle.each_line(END_OF_RECORD) do |raw| 
        begin
          record = MARC::Reader.decode(raw, :forgiving => true)
          yield record 
        rescue StandardError => e
          # caught exception just keep barrelling along
          # TODO add logging
        end
      end
    end
